            <input type="number" id="poles" placeholder="Ingresa la cantidad de palos (n)" required>
            <button type="button" onclick="calculateForce()">Calcular</button>

        <div class="result" id="result"></div>

    <script>
        function calculateForce() {
            const weight = parseFloat(document.getElementById('weight').value);
            const poles = parseInt(document.getElementById('poles').value);

            if (weight > 0 && poles > 0) {
                const force = weight / poles;
                document.getElementById('result').textContent = `La fuerza en cada palo es: ${force.toFixed(2)} N`;
            } else {
                document.getElementById('result').textContent = 'Por favor, ingresa valores v√°lidos.';
            }
        }
    </script>
